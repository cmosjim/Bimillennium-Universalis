			is_subject = no

			development = 4

		always = yes
	}

			factor = 0.1
			scale = yes

			scale = yes

				tooltip = tooltip_any_neighbor_has_embraced_legalism

					NOT = { owned_by = ROOT }

				}
			}
		}

			factor = 1
			scale = yes
			custom_trigger_tooltip = {
				tooltip = tooltip_capital_empire
				is_capital = yes
				owner = { government_rank = 3 }
			}
		}

			factor = 0.5
			scale = yes
			custom_trigger_tooltip = {
				tooltip = tooltip_province_empire_12_development

			factor = 0.5
			scale = yes
			custom_trigger_tooltip = {
				tooltip = tooltip_adm_tech14_nontribal_capital
				is_capital = yes
				owner = { 
					adm_tech = 8

					NOT = { government = native }
				}
			}
		}

			factor = 0.5
			scale = yes
			custom_trigger_tooltip = {
				tooltip = tooltip_adm_tech14_capital
