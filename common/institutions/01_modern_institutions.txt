# Imperialism
#######################################
imperialism = {
	penalty = 0.5

	bonus = {
		core_creation = -0.20
	}
	
	historical_start_date = 1750.1.1
	historical_start_province = 236 #London

	history = {
		OR = {
			AND = {
				is_year = 1750
				continent = europe
				owner = { num_of_cities = 50 }
			}
			AND = {
				is_year = 1800
				owner = { technology_group = western }
			}
			AND = {
				is_year = 1850
				owner = { capital_scope = { continent = europe } }
			}
			AND = {
				is_year = 1900
				culture = japanese
			}
			AND = {
				is_year = 1945
				always = yes
			}
		}
	}
	
	can_start = {
		is_year = 1750
		
		is_island = no
		is_capital = yes
		owner = { 
			num_of_cities = 50
			any_owned_province = { continent = europe }
		}
	}
	start_chance = 10		#in 100 for when a province can_start
	
	
	on_start = et_institutions.1
	
	
	can_embrace = {
		always = yes
	}

	
	embracement_speed = {
		modifier = {
			factor = 0.1
			scale = yes
			custom_trigger_tooltip = {
				tooltip = tooltip_any_friendly_coast_border_province_has_embraced_imperialism
				any_friendly_coast_border_province = {
					imperialism = 100
				}
			}
		}
		modifier = {
			factor = 0.1
			scale = yes
			
			custom_trigger_tooltip = {
				tooltip = tooltip_any_neighbor_has_embraced_imperialism
				any_neighbor_province = {
					NOT = { owned_by = ROOT }
					imperialism = 100
				}
			}
		}
		modifier = {
			factor = 5
			scale = yes
			continent = europe
			is_capital = yes
			owner = { any_owned_province = { NOT = { continent = europe } } }
		}
		modifier = {
			factor = 5
			custom_trigger_tooltip = {
				tooltip = tooltip_full_expansion_and_influence
				is_capital = yes
				owner = {
					full_idea_group = expansion_ideas
					full_idea_group = influence_ideas
				}
			}
			any_province = {
				is_institution_origin = enlightenment
				continent = PREV
			}
		}
		modifier = {
			factor = 1
			custom_trigger_tooltip = {
				tooltip = tooltip_full_expansion_and_influence
				is_capital = yes
				owner = {
					full_idea_group = expansion_ideas
					full_idea_group = influence_ideas
				}
			}
			any_province = {
				is_institution_origin = enlightenment
				NOT = { continent = PREV }
			}
		}
		modifier = {
			factor = 5
			potential = {
				owner = {
					is_colonial_nation = yes
				}
			}
			custom_trigger_tooltip = {
				tooltip = tooltip_colonial_overlord_imperialism
				OR = {
					has_port = yes
					is_capital = yes
				}
				owner = {
					is_colonial_nation = yes
					overlord = { has_institution = imperialism }
				}
			}
		}
		modifier = {
			factor = 2
			scale = yes
			potential = {
				owner = {
					has_institution = imperialism
				}
			}
			custom_trigger_tooltip = {
				tooltip = tooltip_imperialism_embraced
				owner = {
					has_institution = imperialism
				}
			}
		}
	}
	
	ai_will_do = { #Cost of embracing institution must be lower than this factor * last month's regular income, and higher if loans are needed. (Or it has a more than long term goal...)
		factor = 24
		
		modifier = {
			factor = 0.25
			is_at_war = yes
		}
		modifier = {
			factor = 2
			neighbors_tech_discount = 0.1
		}
		modifier = {
			factor = 2
			neighbors_tech_discount = 0.2
		}
		modifier = {
			factor = 2
			neighbors_tech_discount = 0.3
		}
		modifier = {
			factor = 2
			neighbors_tech_discount = 0.4
		}
	}
}
