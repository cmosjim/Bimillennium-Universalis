#appear in the world. which from random events from on_actions.
country_event = {
	id = rab_treasury_events.7
	title = "rab_treasury_events.NAME7"
	desc = "rab_treasury_events.DESC7"
	picture = treasury_stopwarflag_eventPicture

	is_triggered_only = yes 
	hidden = yes
	
	trigger = {
		NOT = { has_global_flag = flg_stopwarflag_3limit }
		NOT = { 
			any_province = {
				has_province_modifier = md_stopwarflag
			}
		}
		any_province = {
			base_tax = 2	
		}	
	}	
		
	option = {
		name = "rab_treasury_events.OPTA7"
		#below sequency is important.
		if = {
			limit = { 
				has_global_flag = flg_stopwarflag_2limit 
			}
			clr_global_flag = flg_stopwarflag_2limit
			set_global_flag = flg_stopwarflag_3limit
		}
		if = {
			limit = { 
				has_global_flag = flg_stopwarflag_1limit 
			}
			clr_global_flag = flg_stopwarflag_1limit
			set_global_flag = flg_stopwarflag_2limit
		}			
		if = {
			limit = { 
				NOT = { 
					OR = {
						has_global_flag = flg_stopwarflag_1limit 
						has_global_flag = flg_stopwarflag_2limit 
						has_global_flag = flg_stopwarflag_3limit 
					}
				}
			}
			set_global_flag = flg_stopwarflag_1limit
		}
	
		random_province = {
			limit = {
				base_tax = 2
				is_sea = no
				is_core = owner
				NOT = { has_province_modifier = md_stopwarflag }
			}	
			add_province_modifier = {
				name = "md_stopwarflag"
				duration = 365
			}
			set_province_flag = flg_pov_md_stopwarflag
			save_global_event_target_as = pov_md_stopwarflag 
		}
	}
}

country_event = {
	id = rab_treasury_events.8
	title = "rab_treasury_events.NAME8"
	desc = "rab_treasury_events.DESC8"
	picture = treasury_stopwarflag_eventPicture

	trigger = {
		any_owned_province = {
			NOT = { has_province_modifier = md_stopwarflag }
			has_province_flag = flg_pov_md_stopwarflag
		}	
	}	
	
	mean_time_to_happen = {
		days = 5
	}	
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					has_global_flag = flg_stopwarflag_1limit
				}
				save_global_event_target_as = tag1_md_stopwarflag
				set_country_flag = flg_tag1_md_stopwarflag
			}	
			if = {
				limit = {
					has_global_flag = flg_stopwarflag_2limit
				}
				save_global_event_target_as = tag2_md_stopwarflag
				set_country_flag = flg_tag2_md_stopwarflag
			}	

			if = {
				limit = {
					has_global_flag = flg_stopwarflag_3limit
				}
				save_global_event_target_as = tag3_md_stopwarflag
				set_country_flag = flg_tag3_md_stopwarflag
			}	
			random_owned_province = {
				limit = {
					has_province_flag = flg_pov_md_stopwarflag 
					NOT = { has_province_modifier = md_stopwarflag }
				}
				clr_province_flag = flg_pov_md_stopwarflag
			}
			
			clear_global_event_target = pov_md_stopwarflag
			
			change_variable = {
				which = num_of_stopwarflag
				value = 1
			}			
		}		
	}
	
	option = {
		name = "rab_treasury_events.OPTA8"
		custom_tooltip = tp_addnum_of_stopwarflag
	}
}
