							id = erik_thorvaldsson_prov_move_events.2
						}
					}
				}					
				if = { 
					limit = { 
						owner = {
							OR = {
								tag = JA1
								tag = JAP
							}	
						}		
					}
					remove_province_modifier = pov_erik_thorvaldsson
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_erik_thorvaldsson"
							duration = -1
						}					
					}
				}
			}	
			if = {
				limit = { 
					is_year = 1003 
				}	
				remove_province_modifier = pov_erik_thorvaldsson		
			}
		}
	}
