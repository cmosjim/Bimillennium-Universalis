country_event = {
	id = rab_acient.4
	title = "et.EVTNAME63"
	desc = "et.EVTDESC63"
	picture = RELIGION_eventPicture
	
	hidden = yes
	
	trigger = {
	
		is_at_war = no
		is_subject = no
		ai = yes
		OR = {
			has_reform = ancient_chinese_warlord_reform
			has_reform = celestial_empire
		}
		mil_power = 50
		#core_percentage = 0.8
		num_of_cities = 4
		stability = 1		
				
		NOT = { war_exhaustion = 8 }
		NOT = { num_of_rebel_armies = 1 }
		NOT = { truce_with = TGG }
		NOT = { alliance_with = TGG }
		army_size = 7
		army_size = TGG
		is_neighbor_of = TGG
		TGG = { NOT = { reverse_has_opinion = { who = ROOT value = 0 } } }	
	}

	mean_time_to_happen = {
		days = 240
	}
	
	option = {
		name = "rab_acient.OPTA4"
			declare_war_with_cb = {
				who = TGG
				casus_belli = cb_chinese_unification
			}
	}
	
}
