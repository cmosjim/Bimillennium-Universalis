country_event = {
	id = recruit_special_units.1
	title = "recruit_special_units.NAME1"
	desc = "recruit_special_units.DESC1"
	picture = RIDING_FIGHTING_eventPicture

	is_triggered_only = yes

	option = {
		name = "recruit_special_units.OPTA1"		
		trigger = {
			manpower = 1
		}		
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				NOT = { treasury = 60 }
			}
			modifier = {
				factor = 0
				cavalry_fraction = 0.45
			}
			modifier = {
				factor = 0
				check_variable = {
					which = var_recruit_num
					value = 10
				}	
			}
		}
		if = {
			limit = {
				variable_arithmetic_trigger = {
					export_to_variable = {
						variable_name = var_land_forcelimit
						value = land_forcelimit
					}
					export_to_variable = {
						variable_name = var_army_size
						value = army_size
					}
					change_variable = {
						which = var_army_size
						which = var_recruit_num
					}
					check_variable = {
						which = var_army_size
						which = var_land_forcelimit
					}
				}				
			}	
			country_event = {
				id = recruit_special_units.999
			}				
			else = {
				random_owned_province = {
					limit = {
						has_building = special_mil_research
					}	
					add_unit_construction = {
						type = special_haw_par_riding
						amount = 1
						speed = 1
						cost = 10
					}
				}
				hidden_effect = {
					change_variable = {
						which = var_recruit_num
						value = 1
					}
					add_manpower = -1					
					country_event = {
						id = recruit_special_units.101
					}
				
					every_neighbor_country = {
						limit = {
							NOT = { has_country_modifier = md_special_haw_par_riding }
							ai = no 
						}	
						country_event = {
							id = recruit_special_units.201
							days = 20
						}	
					}	
				}
			}	
		}
	}
	option = {
		name = "recruit_special_units.OPTB1"
		ai_chance = { factor = 1 }
		hidden_effect = {
			clr_country_flag = tmp_recruit_haw_par_riding
			# var_recruit_num = 0 
			
			
			
		}		
	}	
}

	id = recruit_special_units.9
	title = "recruit_special_units.NAME9"
	desc = "recruit_special_units.DESC9"
	picture = RIDING_FIGHTING_eventPicture

	is_triggered_only = yes

	option = {
		name = "recruit_special_units.OPTA9"
		trigger = {
			manpower = 1
		}		
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				NOT = { treasury = 60 }
			}
			modifier = {
				factor = 0
				cavalry_fraction = 0.33
			}	
			modifier = {
				factor = 0
				check_variable = {
					which = var_recruit_num
					value = 10
				}	
			}
		}	
		if = {
			limit = {
				variable_arithmetic_trigger = {
					export_to_variable = {
						variable_name = var_land_forcelimit
						value = land_forcelimit
					}
					export_to_variable = {
						variable_name = var_army_size
						value = army_size
					}
					change_variable = {
						which = var_army_size
						which = var_recruit_num
					}
					check_variable = {
						which = var_army_size
						which = var_land_forcelimit
					}
				}				
			}
			country_event = {
				id = recruit_special_units.999
			}
			else = {
				random_owned_province = {
					limit = {
						has_building = special_mil_research
					}	
					add_unit_construction = {
						type = special_baiweiqi_riding
						amount = 1
						speed = 1
						cost = 11
					}
				}
				hidden_effect = {
					change_variable = {
						which = var_recruit_num
						value = 1
					}
					add_manpower = -1
					country_event = {
						id = recruit_special_units.109
					}
					every_neighbor_country = {
						limit = {
							NOT = { has_country_modifier = md_special_baiweiqi_riding }
							ai = no 
						}	
						country_event = {
							id = recruit_special_units.209
							days = 20
						}
					}					
				}				
			}	
		}			
	}
	option = {
		name = "recruit_special_units.OPTB9"
		ai_chance = { factor = 1 }
		hidden_effect = {
			clr_country_flag = tmp_recruit_special_baiweiqi_riding
		}		
	}
}
